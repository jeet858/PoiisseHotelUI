import React, { useState, useEffect } from 'react';
import { Form, Col, Row, Card, Button } from 'react-bootstrap';
import { connect } from 'react-redux';

import {
    set_fname,
    set_lname,
    set_email,
    set_address1,
    set_address2,
    set_city,
    set_dist,
    set_state,
    set_country, set_pin, set_addressType, set_addressId
} from '../../actions/UserAction';
import { set_special_request } from '../../actions/CheckoutAction';


const Address = (props) => {
    
    const [formData, setFormData] = useState({
        addrid:-1,
        email: "",
        fname: "",
        lname: "",
        address1: "",
        address2: "",
        city: "",
        dist: "",
        state: "",
        pin: "",
        rqst: "",
        type: ""
    })

    const [validated, setValidated] = useState(false);
    const [formVerified, setFormVerified] = useState(true);
    const [addressSelected, setAddressSelected] = useState(false);
    const [customerDetails, setCustomerDetails] = useState([]);

    const { email, fname, lname, address1, address2, city, state, pin, rqst, dist, type } = formData;
    
    const selectAddressType = (e) => {
        setFormData({...formData, type: e.target.value})
        //console.log(e.target.value)
    }
    const handleOnclick = (event) => {
        
        const form = event.currentTarget;
        console.log("handleOnclick ", form.checkValidity());

        if (form.checkValidity() === false) {
            event.stopPropagation();
        }

        setValidated(true);
        event.preventDefault();
        if (form.checkValidity() === true) {
            const copyDetails = [...customerDetails]
            copyDetails.push(formData)
            setCustomerDetails(copyDetails)
            setFormVerified(true)
        }
    }

    const setFinalAddress = (item) => {
        //+++++++++++++++++++++++++
        props.set_fname(item.fname);
        props.set_lname(item.lname);
        props.set_email(item.email);
        props.set_address1(item.address1);
        props.set_address2(item.address2);
        props.set_city(item.city);
        props.set_state(item.state);
        props.set_dist(item.dist);
        props.set_pin(item.pin);
        props.set_special_request(item.rqst);
        props.set_addressType(item.type);
        props.set_addressId(item.addrid)
        //+++++++++++++++++++++++++
    }
    const selectAddress = (data) => {
        setFormData(data)
        setFinalAddress(data);
        setAddressSelected(true);
        
        setTimeout(() => { props.setAddressCard(false); props.setDeliveryCard(true); }, 500);
    }
    
    /**
     *@function
     * only for development purpose 
     */
    /* useEffect(() => {
        if(props.debug) {
            props.setAddressCard(false); 
            props.setDeliveryCard(true);
        }
    }, [props.debug]); */
    
    const selectThisAddress = (address) => {
        setFormData(address)
    }

    /* useEffect(()=>{
        if(props.customerDetails.length === 0) {
            setFormVerified(false)
        }
        else {
            setCustomerDetails(props.customerDetails)
            setFormVerified(true)
        }
    },[props.customerDetails]) */
    
    useEffect(() => {
        handleOnChange({target:{name:"pin", value:props.pincode}})
        // eslint-disable-next-line 
    },[props.pincode])

    const renderAddresses = () => {
        let list = [];
        if(props.customerDetails.length>0){
            list = props.customerDetails.map((item, i) => {
                return (
                    <Col md="6" className="pl-0 mb-1" key={i}>
                        <Card>
                            <Card.Header>
                                <p>
                                    <i style={{ cursor: "pointer" }} onClick={()=>addressChangeHandler(item,i)} className="fa fa-pencil float-right"></i>
                                </p>
                            </Card.Header>
                            <Card.Body>
                                    
                                <Row>
                                    <Col className="pl-0">{`${item.fname} ${item.lname}, ${item.address1}, ${item.city}, ${item.state}, ${item.pin}`}</Col>
                                </Row>
                                    
                            </Card.Body>
                            <Card.Footer>
                                <Button onClick={() => selectAddress(item)}>Deliver Here</Button>
                            </Card.Footer>
                        </Card>
                    </Col>
                )
            })
       }
       
        return list; 
    } 
    const handleOnChange = (e) => {
        setFormData({ ...formData, [e.target.name]: e.target.value })
    }
    const addAddressHandler = () => {
        handleOnChange({target:{name:"pin", value:props.pincode}})
        setFormVerified(false);
    }
    const addressChangeHandler = (data,i) => {
        console.log("addressChangeHandler", data);
        const copyDetails = [...customerDetails]
        copyDetails.splice(i,1)
        setCustomerDetails(copyDetails)
        selectThisAddress(data)
        setFormVerified(false);
        props.setAddressCard(true);
    }

    const isReturningUser = () => {
        return (props.customerDetails.length>0) ? true : false

                        : <label className="number-before-check">2</label>
                    } &nbsp; Address <button className="float-right btn btn-link" onClick={addAddressHandler}> + Add New Address</button>
                    {isReturningUser() ?
                        <div>
                            <Row>
                                <Col md="6 pl-0 mb-1">
                                    <Row>
                                        <Col className="pl-0">{`${fname} ${lname}, ${address1}, ${city}, ${state}, ${pin}`}</Col>
                                    </Row>
                                </Col>
                            </Row>
                        </div> : ""}
                </Card.Header>
                <Card.Body style={{ display: isReturningUser() ? "none" : "block" }}>
                {isReturningUser() ?
                    <Form noValidate validated={validated} onSubmit={handleOnclick} className="mb-0">
                        <Form.Row>
                            <Form.Group as={Col} >
                                <Form.Label>First Name</Form.Label>
                                <Form.Control placeholder="First Name" value={fname}
                                    name="fname" onChange={handleOnChange} required />
                            </Form.Group>

                            <Form.Group as={Col} >
                                <Form.Label>Last Name</Form.Label>
                                <Form.Control placeholder="Last Name" value={lname}
                                    name="lname" onChange={handleOnChange} required />
                            </Form.Group>
                        </Form.Row>
                        <Form.Group>
                            <Form.Label>Email</Form.Label>
                            <Form.Control type="email" placeholder="Enter Email" value={email}
                                name="email" onChange={handleOnChange} required />
                        </Form.Group>
                        <Form.Group >
                            <Form.Label>Address Line 1</Form.Label>
                            <Form.Control placeholder="Number & Street" value={address1}
                                name="address1" onChange={handleOnChange} required />
                        </Form.Group>

                        <Form.Group >
                            <Form.Label>Address Line 2</Form.Label>
                            <Form.Control placeholder="Number & Street" value={address2}
                                name="address2" onChange={handleOnChange} />
                        </Form.Group>

                        <Form.Row>
                            <Form.Group as={Col}>
                                <Form.Label>City</Form.Label>
                                <Form.Control value={city}
                                    name="city" onChange={handleOnChange} required />
                            </Form.Group>
                            <Form.Group as={Col}>
                                <Form.Label>District</Form.Label>
                                <Form.Control value={dist}
                                    name="dist" onChange={handleOnChange} required />
                            </Form.Group>
                        </Form.Row>
                        <Form.Row>
                            <Form.Group as={Col}>
                                <Form.Label>State</Form.Label>
                                <Form.Control value={state}
                                    name="state" onChange={handleOnChange} required />
                            </Form.Group>

                            <Form.Group as={Col}>
                                <Form.Label>PIN</Form.Label>
                                <Form.Control value={pin}
                                    name="pin" readOnly type="number" required />
                            </Form.Group>
                        </Form.Row>
                        <Form.Group>
                            <Form.Label>Special Request</Form.Label>
                            <Form.Control as="textarea" name="rqst" value={rqst} onChange={handleOnChange} rows="3" />
                        </Form.Group>
                        <Form.Group>
                            <Form.Label>Address Type</Form.Label>
                            <Form.Control value={type} onChange={selectAddressType}

                        <Col md="12 text-right p-0">
                            <button className="btn btn-success bgpinbutton" type="submit" >Next</button>
                        </Col>
                    </Form> :
                    renderAddresses()}
                </Card.Body>
            </Card>
        </React.Fragment >
    )
}
